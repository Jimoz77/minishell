47char **update_pid(char **env)
48{
49 int index;
50 char *pid_str;
51 char *new_env;
52
53 pid_str = getenv("SYSTEMD_EXEC_PID");
54 if (!pid_str)
55 return (env);
56 index = find_systemd_exec_pid_index(env);
57 if (index != -1)
58 {
59 new_env = ft_strjoin("SYSTEMD_EXEC_PID=", pid_str);
60 free(env[index]);
61 env[index] = new_env;
62 }
63 return (env);
64}
