25int count_matching_files(const char *pattern)
26{
27 DIR *dir;
28 struct dirent *entry;
29 int count;
30
31 dir = opendir(".");
32 if (!dir)
33 return (0);
34 count = 0;
35 entry = readdir(dir);
36 while (entry != NULL)
37 {

40 entry = readdir(dir);
41 continue ;
42 }
43 if (match_pattern(pattern, entry->d_name))

45 entry = readdir(dir);
46 }

133
134 i = 0;
135 while (i < count - 1)

137 j = 0;

141 max_len = ft_strlen(strings[j]);
142 else
143 max_len = ft_strlen(strings[j + 1]);
144 if (ft_strncmp(strings[j], strings[j + 1], max_len) > 0)
145 {
146 temp = strings[j];
147 strings[j] = strings[j + 1];
148 strings[j + 1] = temp;

151 }
