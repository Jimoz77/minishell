108char **process_env_lines(char *buffer)
109{
110 t_process_env_l v;
111
112 v.count = 0;
113 v.current_size = 1;
114 v.env = allocate_initial_env();
115 if (!v.env)
116 return (NULL);
117 v.line = ft_strtok(buffer, '\n');
118 while (v.line)

120 if ((size_t)v.count + 1 >= (size_t)v.current_size)
121 {
122 v.current_size *= 2;

124 if (!v.env)
125 return (NULL);
126 }
127 v.env[v.count] = ft_strdup(v.line);
128 if (!v.env[v.count])

130 v.count++;
131 v.line = ft_strtok(NULL, '\n');
132 }
