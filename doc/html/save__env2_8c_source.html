84void save_history(char *cmd, t_shell *shell)
85{
86 int fd;
87
88 if (should_skip_duplicate(shell->old_cmd, cmd))
89 return ;
90 fd = open_history_file();
91 if (fd == -1)
92 return ;
93 write(fd, cmd, ft_strlen(cmd));
94 write(fd, "\n", 1);
95 close(fd);
96 if (shell->old_cmd)
97 free(shell->old_cmd);
98 shell->old_cmd = ft_strdup(cmd);
