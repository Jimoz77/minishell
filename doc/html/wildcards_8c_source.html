24int contains_wildcard(const char *str)
25{
26 int i;
27 int in_quotes;
28 char quote_type;
29
30 if (!str)
31 return (0);
32 i = 0;
33 in_quotes = 0;
34 quote_type = 0;
35 while (str[i])
36 {
37 if (!in_quotes && (str[i] == '\'' || str[i] == '"'))
38 {
39 in_quotes = 1;
40 quote_type = str[i];
41 }
42 else if (in_quotes && str[i] == quote_type)
43 in_quotes = 0;
44 else if (!in_quotes && str[i] == '*')
45 return (1);
46 i++;
47 }
48 return (0);
49}
