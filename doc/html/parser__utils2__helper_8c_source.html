88static t_node *extract_redirections(t_token **tokens, t_node *cmd_node)
89{
90 t_token *current;
91 t_token *prev;
92 t_token *next;
93
94 if (!tokens || !*tokens || !cmd_node)
95 return (cmd_node);
96 current = *tokens;
97 prev = NULL;
98 while (current)
99 {
100 next = current->next;
101 if (is_redirection(current->type) && current->next
102 && current->next->type == TOKEN_WORD)
103 {
104 process_redirection(current, cmd_node, &prev, tokens);
105 update_after_redirect(&current, &prev, tokens);
106 }
107 else
108 update_normal_token(&current, &prev, next);
109 }
110 return (cmd_node);
111}
