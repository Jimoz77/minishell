126int process_cmd_tokens(t_node *node, char ***envp, t_shell *shell)
127{
128 t_process_cmd_tokens v;
129
130 if (expand_cmd_with_original_tokens(node, shell) == 0)
131 {
132 if (ft_is_builtin(node->cmd, envp))
133 return (exec_builtin_with_redirections(node, envp));
134 else
135 return (exec_cmd_with_redirections(node, *envp, shell));
136 }
137 v.original_tokens = shell->tokens;
138 v.temp_tokens = create_tokens_from_cmd(node->cmd, shell);
139 if (!v.temp_tokens)
140 {
141 shell->tokens = v.original_tokens;
142 return (handle_no_tokens_case(node, envp, shell));
143 }
144 process_cmd_expansion(v.temp_tokens, shell);
145 shell->tokens = v.original_tokens;
146 update_cmd_from_tokens(node->cmd, v.temp_tokens);
147 if (ft_is_builtin(node->cmd, envp))
148 v.result = exec_builtin_with_redirections(node, envp);
149 else
150 v.result = exec_cmd_with_redirections(node, *envp, shell);
151 free_tokens(v.temp_tokens);
152 return (v.result);
153}
